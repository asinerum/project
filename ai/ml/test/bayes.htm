Usage:
//[data] is an 1D array of length D of strings
var data =
[
 "amazing, awesome movie! yeah! oh boy",
 "sweet, this is incredibly, amazing! perfect, great!",
 "terrible, shitty thing! damn! sucks!",
];
//[labels] is an 1D array of length D of any type values
var labels = ["positive", "positive", "negative"];
//Start a NaiveBayes classifier
var nb;
naivebayes(data,labels).then(r=>console.log(nb=r));
//Ask the classifier to predict/categorize something
nb.predict('awesome, cool, amazing').then(console.log); //positive
//Serialize the classifier as a JSON model
var stateJson = nb.toJson();
//Load the classifier back from its JSON model
var nb2 = NB.fromJson(stateJson);
nb2.predict('bitch, what a hell, fuck').then(console.log); //negative
